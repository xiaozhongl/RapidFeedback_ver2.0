/LoginServlet

POST
{
    username: String    // this key is named as email in RegisterServlet
    password:   String
}
response
{
    login_ACK: // >0 id :succeed/ 0: wrong password/ -1: no email address in database/ -2: db/servlet error
    projectList: ArrayList<Project>
    firstName:  String
    token:  String
}
